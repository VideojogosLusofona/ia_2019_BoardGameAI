177  // Setup session depending on how many AIs will play
178  if (activeAIs.Count == 0)
179  {
180  // A game between human players
181  uiWhoPlaysFirst = false;
182  uiBlockStartNextMatch = true;
183  uiBlockShowResult = true;
184  allMatches.Add(new Match(humanPlayer, humanPlayer), Winner.None);
185  }
186  else if (activeAIs.Count == 1)
187  {
188  // A game between a human and an AI
189  uiWhoPlaysFirst = true;
190  uiBlockStartNextMatch = true;
191  uiBlockShowResult = true;
192  allMatches.Add(new Match(humanPlayer, activeAIs[0]), Winner.None);
193  }
194  else if (activeAIs.Count == 2)
195  {
196  // A game between two AIs, ask who plays first
197  uiWhoPlaysFirst = true;
198  uiBlockStartNextMatch = true;
199  uiBlockShowResult = true;
200  allMatches.Add(new Match(activeAIs[0], activeAIs[1]), Winner.None);
201  }
202  else
203  {
204  // Multiple AIs, run a tournament
205  uiWhoPlaysFirst = false;
206  uiBlockStartNextMatch = pressButtonBeforeMatch;
207  uiBlockShowResult = pressButtonAfterMatch;
208 
209  // In this mode we need an even number of players to set up the
210  // matches, so add a fake one if necessary
211  if (activeAIs.Count % 2 != 0) activeAIs.Add(new DummyPlayer());
212 
213  // Setup matches using the round-robin method
214  // https://en.wikipedia.org/wiki/Round-robin_tournament
215  for (int i = 1; i < activeAIs.Count; i++)
216  {
217  // This will be the AI to swap position after each round
218  IPlayer aiToSwapPosition;
219 
220  // Set up matches for current round i
221  for (int j = 0; j < activeAIs.Count / 2; j++)
222  {
223  // This is match j for current round i
224  Match match = new Match(
225  activeAIs[j], activeAIs[activeAIs.Count - 1 - j]);
226  // Only add match to match list if it's not a dummy match
227  if (!(match.player1 is DummyPlayer
228  || match.player2 is DummyPlayer))
229  {
230  // Each match is in practice two matches, so both AIs
231  // can have a match where they are the first to play
232  allMatches.Add(match, Winner.None);
233  allMatches.Add(match.Swapped, Winner.None);
234  }
235  }
236  // Swap AI positions for next round
237  aiToSwapPosition = activeAIs[activeAIs.Count - 1];
238  activeAIs.RemoveAt(activeAIs.Count - 1);
239  activeAIs.Insert(1, aiToSwapPosition);
240  }
241  }
242 
243  // Get the match enumerator and initialize it
244  matchEnumerator = allMatches.Keys.ToList().GetEnumerator();
245  matchEnumerator.MoveNext();
246  currentMatch = matchEnumerator.Current;
247  }

304  // End current match an change session state to PostMatch
305  private void EndCurrentMatchCallback()
306  {
307  // Keep result of current match
308  allMatches[currentMatch] = matchController.Result;
309 
310  // If we are in tournament mode, update standings / classification
311  if (activeAIs.Count > 2)
312  {
313  switch (matchController.Result)
314  {
315  // White won
316  case Winner.White:
317  standings[currentMatch.player1] += pointsPerWin;
318  standings[currentMatch.player2] += pointsPerLoss;
319  break;
320  // Red won
321  case Winner.Red:
322  standings[currentMatch.player2] += pointsPerWin;
323  standings[currentMatch.player1] += pointsPerLoss;
324  break;
325  // Game ended in a draw
326  case Winner.Draw:
327  standings[currentMatch.player1] += pointsPerDraw;
328  standings[currentMatch.player2] += pointsPerDraw;
329  break;
330  // Invalid situation
331  default:
332  throw new InvalidOperationException(
333  "Invalid end of match result");
334  }
335  }
336 
337  // Set session state to PostMatch
338  state = SessionState.PostMatch;
339  }

341  // Destroy old match instance and check if there are more matches to
342  // play. If so, get next match and update state to PreMatch. Otherwise,
343  // dispose of the match enumerator and update session state to End.
344  private void DestroyAndIterateMatchCallback()
345  {
346  // Destroy old match instance
347  Destroy(matchInstance);
348 
349  // Are there more matches to play?
350  if (matchEnumerator.MoveNext())
351  {
352  // If so, get next match and set session state to PreMatch
353  currentMatch = matchEnumerator.Current;
354  state = SessionState.PreMatch;
355  }
356  else
357  {
358  // Otherwise, dispose of the match enumerator and set session state
359  // to End
360  matchEnumerator.Dispose();
361  state = SessionState.End;
362  }
363  }
