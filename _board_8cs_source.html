418  public Winner CheckWinner(Pos[] solution = null)
419  {
420  // If an array for placing the solution was given, make sure it has
421  // the required capacity
422  if (solution != null && solution.Length < piecesInSequence)
423  throw new InvalidOperationException("Array for placing solution " +
424  $"must have at least length {piecesInSequence}");
425 
426  // Is the board full? Then we have a draw
427  if (numMoves == cols * rows || PiecesLeft == 0) return Winner.Draw;
428 
429  // Check for all different pieces
430  foreach (PieceFuncPlayer funcPlayer in pieceFuncsPlayers)
431  {
432  // Check all possible corridors
433  foreach (IEnumerable<Pos> corridor in winCorridors)
434  {
435  // Reset count for this corridor
436  int count = 0;
437 
438  // Check positions in this corridor
439  foreach (Pos p in corridor)
440  {
441  // Update solution array
442  if (solution != null) solution[count] = p;
443 
444  // Does position contain the appropriate piece?
445  if (board[p.col, p.row].HasValue &&
446  funcPlayer.checkPieceFunc(board[p.col, p.row].Value))
447  // Yes it does, increment counter
448  count++;
449  else
450  // No it doesn't, reset count
451  count = 0;
452  // Did we find enough pieces in a row?
453  if (count == piecesInSequence)
454  // If so, return winner
455  return funcPlayer.player == PColor.White
456  ? Winner.White
457  : Winner.Red;
458  }
459  }
460  }
