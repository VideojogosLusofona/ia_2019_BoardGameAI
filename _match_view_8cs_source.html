384  internal void UpdateBoard(Move move, Winner result, Pos[] solution)
385  {
386  // Update finished flag
387  this.finished = result != Winner.None;
388 
389  // Is the screen board position empty and the game board has a piece?
390  if (pieces[move.row, move.col] == null
391  && board[move.row, move.col].HasValue)
392  {
393  // Then also place that piece in the screen board
394 
395  // Reference to the piece prefab
396  GameObject piecePrefab;
397 
398  // The piece on the game board to also put in the screen board
399  Piece piece = board[move.row, move.col].Value;
400 
401  // Determine the piece prefab to use based on the board piece
402  if (piece.Is(PColor.White, PShape.Round))
403  piecePrefab = whiteRoundPiece;
404  else if (piece.Is(PColor.White, PShape.Square))
405  piecePrefab = whiteSquarePiece;
406  else if (piece.Is(PColor.Red, PShape.Round))
407  piecePrefab = redRoundPiece;
408  else if (piece.Is(PColor.Red, PShape.Square))
409  piecePrefab = redSquarePiece;
410  else
411  throw new InvalidOperationException(
412  "Trying to instantiate an invalid piece");
413 
414  // Instantiate the screen piece
415  pieces[move.row, move.col] = Instantiate(
416  piecePrefab,
417  new Vector3(
418  // Horizontal position
419  leftPoleBase.x + move.col * distBtwPoles,
420  // Vertical position
421  leftPoleBase.y
422  + move.row * (totalHeightForPieces / board.rows)
423  + piecesLength / 2,
424  // Z-axis
425  2),
426  Quaternion.identity,
427  transform);
428 
429  // Correct scale of screen piece
430  pieces[move.row, move.col].transform.localScale =
431  piecesScale * Vector3.one;
432 
433  // Is the column now full?
434  if (board.IsColumnFull(move.col))
435  {
436  // If so, close the arrow
437  uiArrows[move.col].Open = false;
438  }
439 
440  // Animation last move
441  StartCoroutine(AnimateLastMove(move));
442  }
443  // Or is the screen board position occupied while the game board
444  // position is empty?
445  else if (pieces[move.row, move.col] != null
446  && !board[move.row, move.col].HasValue)
447  {
448  // In such case, destroy the screen board piece
449  Destroy(pieces[move.row, move.col]);
450  pieces[move.row, move.col] = null;
451 
452  // Open the arrow
453  uiArrows[move.col].Open = true;
454  }
455  // Otherwise it's an impossible situation and we have a bug
456  else
457  {
458  throw new InvalidOperationException(
459  "Board view representation not in sync with board model");
460  }
461 
462  // Update previous player shape choice
463  SelectShape(move.piece.color, move.piece.shape);
464 
465  // Disable or enable GUI stuff depending on who's playing next
466  foreach (HumanMoveButton arrow in uiArrows)
467  {
468  arrow.gameObject.SetActive(

470  }
471 
472  // Notify listeners that board was updated
473  BoardUpdated.Invoke();
474 
475  // If match is finished and result is not a draw, draw a line marking
476  // the solution
477  if (result == Winner.Red || result == Winner.White)
